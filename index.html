int binaryToDecimal (long num) {
    long result = 0;
    char current;
    std::string num_str = std::to_string(num);
                        
    for (int i = 0; i < num_str.length(); i++) { 
        result += ((int) num_str[num_str.length() - i - 1] - '0') * pow(2, i);
    }

    return result;
}
